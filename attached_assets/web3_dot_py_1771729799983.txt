import os
from web3 import Web3
from eth_account import Account

# 1. Initialize Web3 and Bot Wallet
rpc_url = os.getenv("TENDERLY_RPC_URL") # Or Arbitrum Mainnet RPC
w3 = Web3(Web3.HTTPProvider(rpc_url))
bot_private_key = os.getenv("BOT_PRIVATE_KEY")
bot_account = Account.from_key(bot_private_key)

# 2. Contract Addresses (From your Replit Audit)
DAI_DEBT_TOKEN = "0x8619d80FB0141ba7F184CbF22fd724116D9f7ffC"
AAVE_POOL = "0x794a61358D6845594F94dc1DB02A252b5b4814aD"
DELEGATION_MANAGER = "0x..." # Your custom DM contract address

# 3. Load the Debt Token ABI (Minimal ABI for delegationWithSig)
DEBT_TOKEN_ABI = [
    {
        "inputs": [
            {"internalType": "address", "name": "delegator", "type": "address"},
            {"internalType": "address", "name": "delegatee", "type": "address"},
            {"internalType": "uint256", "name": "value", "type": "uint256"},
            {"internalType": "uint256", "name": "deadline", "type": "uint256"},
            {"internalType": "uint8", "name": "v", "type": "uint8"},
            {"internalType": "bytes32", "name": "r", "type": "bytes32"},
            {"internalType": "bytes32", "name": "s", "type": "bytes32"}
        ],
        "name": "delegationWithSig",
        "outputs": [],
        "stateMutability": "nonpayable",
        "type": "function"
    }
]

debt_contract = w3.eth.contract(address=DAI_DEBT_TOKEN, abi=DEBT_TOKEN_ABI)

def activate_credit_and_borrow(user_address: str, raw_signature: str, deadline: int):
    """
    Submits the user's off-chain signature to Aave to unlock credit, 
    then immediately borrows DAI on their behalf.
    """
    # 4. Parse the Signature into v, r, s components
    # The frontend ethers.js signature is a 65-byte hex string
    signature_bytes = w3.to_bytes(hexstr=raw_signature)
    r = signature_bytes[0:32]
    s = signature_bytes[32:64]
    v = signature_bytes[64]
    
    # EIP-712 'v' is usually 27 or 28. Sometimes ethers.js returns 0 or 1, which needs fixing:
    if v < 27:
        v += 27 

    print(f"Submitting gasless delegation for {user_address}...")
    
    # 5. Build the delegationWithSig Transaction
    max_uint = 2**256 - 1
    
    delegate_tx = debt_contract.functions.delegationWithSig(
        delegator=w3.to_checksum_address(user_address),
        delegatee=w3.to_checksum_address(DELEGATION_MANAGER),
        value=max_uint,
        deadline=deadline,
        v=v,
        r=r,
        s=s
    ).build_transaction({
        'from': bot_account.address,
        'nonce': w3.eth.get_transaction_count(bot_account.address),
        # On Arbitrum, gas parameters might need fine-tuning, but this is standard:
        'gas': 200000, 
        'maxFeePerGas': w3.to_wei('0.1', 'gwei'),
        'maxPriorityFeePerGas': w3.to_wei('0.1', 'gwei'),
    })

    # 6. Sign and Broadcast the Delegation
    signed_delegate_tx = w3.eth.account.sign_transaction(delegate_tx, private_key=bot_private_key)
    tx_hash = w3.eth.send_raw_transaction(signed_delegate_tx.rawTransaction)
    
    print(f"Delegation Tx Sent: {tx_hash.hex()}")
    w3.eth.wait_for_transaction_receipt(tx_hash)
    print("Delegation Successful! The agent can now borrow.")
    
    # 7. Next Step: Call your DelegationManager or Aave Pool directly to execute the borrow
    # e.g., execute_borrow_logic(...)
