// Step 3/3: Aave credit delegation
btn.textContent = "Step 3/3: Checking credit delegation...";

const VARIABLE_DEBT_TOKENS = {
  DAI:  "0x8619d80FB0141ba7F184CbF22fd724116D9f7ffC",
  WBTC: "0x92b42c66840C7AD907b4BF74879FF3eF7c529473",
  WETH: "0x0c84331e39d6658Cd6e6b9ba04736cC4c4734351",
  USDC: "0xFCCf3cAbbe80101232d343252614b6A3eE81C989",
  USDT: "0x6ab707Aca953eDAeFBc4fD23bA73294241490620",
};

const DEBT_TOKEN_ABI = [
  "function borrowAllowance(address fromUser, address toUser) view returns (uint256)",
  "function approveDelegation(address delegatee, uint256 amount)"
];

const creditDelegationCalls = [];

for (const [symbol, debtTokenAddr] of Object.entries(VARIABLE_DEBT_TOKENS)) {
  try {
    const debtToken = new ethers.Contract(
      debtTokenAddr, DEBT_TOKEN_ABI, signer
    );
    const allowance = await debtToken.borrowAllowance(
      userWallet, DELEGATION_MANAGER
    );
    if (allowance.eq(0)) {
      creditDelegationCalls.push({
        target: debtTokenAddr,
        allowFailure: true, // allow individual token failures
        callData: debtToken.interface.encodeFunctionData(
          "approveDelegation",
          [DELEGATION_MANAGER, ethers.constants.MaxUint256]
        )
      });
    }
  } catch (e) {
    console.warn(`Credit delegation check failed for ${symbol}: ${e.message}`);
  }
}

if (creditDelegationCalls.length > 0) {
  btn.textContent = 
    `Step 3/3: Approving credit delegation ` +
    `(${creditDelegationCalls.length} tokens, 1 signature)...`;
  
  const multicall = new ethers.Contract(
    MULTICALL3_ADDRESS, MULTICALL3_ABI, signer
  );
  const tx = await multicall.aggregate3(creditDelegationCalls);
  await tx.wait();
  console.log("Credit delegation approved via Multicall3");
} else {
  console.log("All credit delegations already in place, skipping Step 3");
}

btn.textContent = "Step 3/3: Complete. Activating Auto-Pilot...";

## Permission Layers for Full Automation

Three separate on-chain permission layers are 
required before the bot can execute strategies 
on behalf of a user wallet. All three are 
checked by validate_full_automation_ready() 
and must pass before strategy_status is set 
to active.

### Layer 1: DelegationManager flags
Contract : REAADelegationManager
           0x7427370Ab4C311B090446544078c819b3946E59d
Function : approveDelegation(maxSupplyPerTx,
           dailySupplyLimit, true, true, true, true)
Flags    : allowSupply, allowBorrow, 
           allowRepay, allowWithdraw = true
Signed by: User (Step 1/3 of onboarding)

### Layer 2: ERC20 token approvals
Tokens   : DAI, WETH, WBTC, USDC, USDT
Spenders : DelegationManager, Aave Pool,
           Uniswap Router
Amount   : MaxUint256
Signed by: User via Multicall3 (Step 2/3)

### Layer 3: Aave V3 credit delegation
Purpose  : Required for DelegationManager to
