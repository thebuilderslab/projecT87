import requests

def get_paraswap_price(src_token, dest_token, src_amount, network=42161):
    url = f"https://api.paraswap.io/prices"
    
    params = {
        "srcToken": src_token,  # DAI address
        "destToken": dest_token,  # WETH address
        "srcDecimals": 18,
        "destDecimals": 18,
        "amount": src_amount,  # Amount in wei
        "side": "SELL",  # or "BUY" for exact output
        "network": network
    }
    
    response = requests.get(url, params=params)
    price_route = response.json()["priceRoute"]
    return price_route
Step 2: Build Transaction with Proper Calldata
Then, POST the priceRoute to /transactions to get the EXACT calldata format Aave expects:

python
def build_paraswap_transaction(price_route, user_address, network=42161):
    url = f"https://api.paraswap.io/transactions/{network}"
    
    body = {
        "srcToken": price_route["srcToken"],
        "destToken": price_route["destToken"],
        "srcAmount": price_route["srcAmount"],
        "destAmount": price_route["destAmount"],
        "priceRoute": price_route,  # MUST pass exact priceRoute object
        "userAddress": user_address,
        "slippage": 100,  # 1% in basis points
        "ignoreChecks": True,  # Skip balance checks (for Aave flash loan)
        "ignoreGasEstimate": True  # Skip gas estimation
    }
    
    response = requests.post(url, json=body)
    tx_data = response.json()
    
    return {
        "to": tx_data["to"],  # Augustus Swapper contract
        "data": tx_data["data"],  # THIS is the calldata you need!
        "value": tx_data["value"],
        "gasPrice": tx_data.get("gasPrice", ""),
        "chainId": tx_data.get("chainId", 42161)
    }

5. Why This Fixes Your Issue​

import requests
from web3 import Web3

class ParaSwapAaveIntegration:
    def __init__(self, w3: Web3):
        self.w3 = w3
        self.api_base = "https://api.paraswap.io"
        self.network = 42161  # Arbitrum
        
    def swap_debt(self, from_debt, to_debt, amount, user_address):
        """Execute Aave debt swap using ParaSwap API"""
        
        # 1. Get optimal price route
        price_route = self._get_price_route(
            src_token=to_debt,
            dest_token=from_debt,
            amount=amount
        )
        
        # 2. Build transaction calldata
        tx_data = self._build_transaction(
            price_route=price_route,
            user_address=user_address
        )
        
        # 3. Execute Aave debt swap with ParaSwap calldata
        return self._execute_debt_swap(
            debt_asset=from_debt,
            new_debt_asset=to_debt,
            amount=amount,
            paraswap_data=tx_data["data"],
            max_new_debt=price_route["srcAmount"]
        )
    
    def _get_price_route(self, src_token, dest_token, amount):
        url = f"{self.api_base}/prices"
        params = {
            "srcToken": src_token,
            "destToken": dest_token,
            "amount": str(amount),
            "srcDecimals": 18,
            "destDecimals": 18,
            "side": "SELL",
            "network": self.network
        }
        response = requests.get(url, params=params)
        return response.json()["priceRoute"]
    
    def _build_transaction(self, price_route, user_address):
        url = f"{self.api_base}/transactions/{self.network}"
        body = {
            "priceRoute": price_route,  # CRITICAL: Pass as-is!
            "srcToken": price_route["srcToken"],
            "destToken": price_route["destToken"],
            "srcAmount": price_route["srcAmount"],
            "destAmount": price_route["destAmount"],
            "userAddress": user_address,
            "slippage": 100,
            "ignoreChecks": True,
            "ignoreGasEstimate": True
        }
        response = requests.post(url, json=body)
        return response.json()
