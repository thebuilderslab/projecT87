import os
import difflib
import shutil
import subprocess
import re
from collections import defaultdict

# Keywords to group files by functionality
KEYWORDS = [
    "diagnostic", "compliance", "launcher", "dashboard",
    "sync", "test", "emergency", "health_monitor", "fetcher"
]

BASE_DIR = "."
ARCHIVE_DIR = "archive_duplicates"
REPORT_FILE = "system_audit_report.txt"

def scan_files():
    py_files = []
    for root, _, files in os.walk(BASE_DIR):
        for f in files:
            if f.endswith(".py"):
                py_files.append(os.path.join(root, f))
    return py_files

def group_by_keyword(files):
    groups = defaultdict(list)
    for f in files:
        name_lower = os.path.basename(f).lower()
        matched = False
        for kw in KEYWORDS:
            if kw in name_lower:
                groups[kw].append(f)
                matched = True
                break
        if not matched:
            groups["misc"].append(f)
    return groups

def git_latest_commit_date(file_path):
    try:
        result = subprocess.run(
            ["git", "log", "-1", "--format=%ct", "--", file_path],
            capture_output=True, text=True, check=True)
        timestamp = int(result.stdout.strip())
        return timestamp
    except Exception:
        # fallback to file modification time if git info unavailable
        return int(os.path.getmtime(file_path))

def choose_canonical(files):
    # Pick the file with latest git commit or fallback mod time (descending order)
    ordered = sorted(files, key=git_latest_commit_date, reverse=True)
    return ordered[0], ordered[1:]  # Return canonical and others

def merge_missing_code(canonical, old_file):
    with open(canonical, "r", encoding="utf-8") as cf, open(old_file, "r", encoding="utf-8") as of:
        canonical_lines = cf.readlines()
        old_lines = of.readlines()

    # Identify lines in old file not present in canonical (simple approach)
    additions = [line for line in old_lines if line.strip() and line not in canonical_lines]

    if additions:
        with open(canonical, "a", encoding="utf-8") as cf:
            cf.write(f"\n\n# --- Merged from older version: {os.path.basename(old_file)} ---\n")
            cf.writelines(additions)
        return True
    return False

def archive_file(file_path):
    os.makedirs(ARCHIVE_DIR, exist_ok=True)
    dst = os.path.join(ARCHIVE_DIR, os.path.basename(file_path))
    count = 1
    while os.path.exists(dst):
        base, ext = os.path.splitext(dst)
        dst = f"{base}_{count}{ext}"
        count += 1
    shutil.move(file_path, dst)

def refactor_imports(canonical_map):
    # canonical_map: { old_file_path: canonical_file_path }
    # Update imports referring to old files to canonical ones
    for root, _, files in os.walk(BASE_DIR):
        for f in files:
            if f.endswith(".py"):
                path = os.path.join(root, f)
                with open(path, "r", encoding="utf-8") as file:
                    content = file.read()

                original_content = content

                for old, canonical in canonical_map.items():
                    old_mod = os.path.splitext(os.path.basename(old))[0]
                    canonical_mod = os.path.splitext(os.path.basename(canonical))

                    # Replace from imports
                    content = re.sub(rf'\bfrom\s+{re.escape(old_mod)}\b', f'from {canonical_mod}', content)
                    # Replace plain imports
                    content = re.sub(rf'\bimport\s+{re.escape(old_mod)}\b', f'import {canonical_mod}', content)

                if content != original_content:
                    with open(path, "w", encoding="utf-8") as file:
                        file.write(content)

def static_syntax_check(files):
    errors = []
    for file_path in files:
        try:
            subprocess.check_output(["python", "-m", "py_compile", file_path], stderr=subprocess.STDOUT)
        except subprocess.CalledProcessError as e:
            err_output = e.output.decode(errors='replace')
            first_line = err_output.strip().splitlines()[0] if err_output else "Unknown Error"
            errors.append(f"{file_path}: {first_line}")
    return errors

def find_import_errors(archived_files):
    errors = []
    archived_mods = {os.path.splitext(os.path.basename(f)) for f in archived_files}
    for root, _, files in os.walk(BASE_DIR):
        for f in files:
            if f.endswith(".py"):
                path = os.path.join(root, f)
                try:
                    with open(path, 'r', encoding='utf-8') as file:
                        content = file.read()
                except:
                    continue
                for mod in archived_mods:
                    import_pattern = re.compile(rf'\b(import|from)\s+{re.escape(mod)}\b')
                    if import_pattern.search(content):
                        errors.append(f"{path}: import of archived module {mod} detected")
    return errors

def main():
    report_lines = []
    py_files = scan_files()
    groups = group_by_keyword(py_files)
    canonical_map = {}
    all_canonicals = []
    archived_files = []

    # Step 1-3: Select canonicals, merge and archive duplicates
    for kw, files in groups.items():
        if len(files) < 2:
            # Single file - treat as canonical with no duplicates
            report_lines.append(f"Feature: {kw} (single file)")
            report_lines.append(f"  Files: {files[0]}")
            all_canonicals.append(files)
            continue

        canonical, duplicates = choose_canonical(files)
        all_canonicals.append(canonical)
        report_lines.append(f"Feature: {kw}")
        report_lines.append(f"  Files: {canonical} (canonical)")
        merged_any = False

        for old_file in duplicates:
            merged = merge_missing_code(canonical, old_file)
            archive_file(old_file)
            archived_files.append(old_file)
            canonical_map[old_file] = canonical
            if merged:
                merged_any = True
                report_lines.append(f"  Merged and archived: {old_file}")
            else:
                report_lines.append(f"  Archived (no merge needed): {old_file}")

        if not duplicates:
            merged_any = False

    # Step 4: Refactor imports
    refactor_imports(canonical_map)

    # Step 5: Static syntax checks on all canonicals
    syntax_errors = static_syntax_check(all_canonicals)

    # Step 6: Check for imports referencing archived files
    import_errors = find_import_errors(archived_files)

    # Step 7: Attempt lightweight test run if tests folder or main.py exists
    test_errors = []
    try:
        # Prefer pytest if available
        subprocess.run(["pytest", "-q"], check=True)
    except FileNotFoundError:
        report_lines.append("Warning: pytest not found, skipping tests")
    except subprocess.CalledProcessError:
        test_errors.append("Tests failed (see pytest output)")

    # Output final report
    report_lines.append("\nErrors:")
    for err in syntax_errors:
        report_lines.append(f"SyntaxError: {err}")
    for err in import_errors:
        report_lines.append(f"ImportError: {err}")
    for err in test_errors:
        report_lines.append(f"TestError: {err}")

    report_text = "\n".join(report_lines)
    with open(REPORT_FILE, "w", encoding="utf-8") as report_file:
        report_file.write(report_text)

    print(report_text)
    print(f"\nAudit complete. Report saved to {REPORT_FILE}")

if __name__ == "__main__":
    main()
