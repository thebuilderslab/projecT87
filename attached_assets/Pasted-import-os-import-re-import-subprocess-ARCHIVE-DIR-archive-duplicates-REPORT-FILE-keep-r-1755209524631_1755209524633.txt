import os
import re
import subprocess

ARCHIVE_DIR = "archive_duplicates"
REPORT_FILE = "keep_remove_report.txt"
VER_REPORT = "verification_report.txt"

def main():
    results = []

    # 1. Check archive exists
    if os.path.isdir(ARCHIVE_DIR):
        archived_files = os.listdir(ARCHIVE_DIR)
        results.append(f"✅ Archive dir exists: {len(archived_files)} files archived")
    else:
        results.append("❌ Archive dir not found")
        archived_files = []

    # 2. Verify canonical merge markers
    if os.path.exists(REPORT_FILE):
        with open(REPORT_FILE) as f:
            report = f.read().splitlines()
        for line in report:
            if "Canonical:" in line:
                canonical_path = line.split("Canonical:")[-1].strip()
                if os.path.exists(canonical_path):
                    with open(canonical_path) as cf:
                        content = cf.read()
                    if "# --- Merged from older version" in content:
                        results.append(f"✅ Merged content found in {canonical_path}")
                    else:
                        results.append(f"⚠️ No merge marker in {canonical_path} (check if no merge was needed)")
    else:
        results.append("❌ keep_remove_report.txt missing")

    # 3. Check imports referencing archived files
    bad_imports = []
    for root, _, files in os.walk("."):
        for name in files:
            if name.endswith(".py"):
                with open(os.path.join(root, name), "r") as f:
                    content = f.read()
                for af in archived_files:
                    mod_name = os.path.splitext(af)[0]
                    if re.search(rf"\b(import|from)\s+{re.escape(mod_name)}\b", content):
                        bad_imports.append(f"{name} -> {mod_name}")
    if bad_imports:
        results.append("❌ Found imports to archived files: " + ", ".join(bad_imports))
    else:
        results.append("✅ No imports to archived files")

    # 4. Run syntax check
    syntax_ok = True
    for root, _, files in os.walk("."):
        for name in files:
            if name.endswith(".py"):
                try:
                    subprocess.check_call(["python", "-m", "py_compile", os.path.join(root, name)],
                                          stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
                except subprocess.CalledProcessError:
                    syntax_ok = False
                    results.append(f"❌ Syntax error in {os.path.join(root, name)}")
    if syntax_ok:
        results.append("✅ Syntax check passed")

    # 5. Optionally run tests
    try:
        subprocess.check_call(["pytest", "-q"], stdout=subprocess.DEVNULL)
        results.append("✅ Tests passed")
    except FileNotFoundError:
        results.append("⚠️ pytest not found, skipping tests")
    except subprocess.CalledProcessError:
        results.append("❌ Tests failed")

    with open(VER_REPORT, "w") as vf:
        vf.write("\n".join(results))

    print("\n".join(results))
    print(f"\nVerification report saved to {VER_REPORT}")

if __name__ == "__main__":
    main()
