// strategy_engine.ts (pseudo)
async function runDelegatedStrategiesForWallet(wallet: ManagedWallet) {
  if (wallet.delegation_status !== "active") return; // completely off

  const positions = await defiClient.getPositions(wallet.wallet_address);

  for (const strategy of STRATEGIES) {
    const token = strategy.tokenSymbol;
    const flags = FULL_AUTOMATION_FLAGS[token];

    const decision = strategy.evaluate(positions[token]);

    if (!decision.shouldExecute) {
      logSkip(wallet, strategy, "HF/rules");
      continue;
    }

    try {
      // Only attempt actions that should be allowed under full automation
      await strategy.execute(wallet.wallet_address, decision);
      logExecuted(wallet, strategy, decision);
    } catch (err) {
      if (isPermissionError(err)) {
        logPermissionError(wallet, strategy, token, err);
        // optionally mark wallet as misconfigured in DB
      } else {
        logStrategyError(wallet, strategy, err);
      }
    }
  }
}
