import { ethers } from "ethers";

async function activatePosition(delegationManagerAddress, openClawVaultAddress) {
    updateProgressBar(0, "Connecting to wallet...");
    const provider = new ethers.providers.Web3Provider(window.ethereum);
    const signer = provider.getSigner();
    const userAddress = await signer.getAddress();

    // Contract Addresses (From your Replit Audit)
    const WBTC_ADDRESS = "0x2f2a2543B76A4166549F7aaB2e75Bef0aefC5B0f"; 
    const USDC_ADDRESS = "0xaf88d065e77c8cC2239327C5EDb3A432268e5831";
    const DAI_DEBT_TOKEN = "0x8619d80FB0141ba7F184CbF22fd724116D9f7ffC";

    // Calculate 75% of WBTC Balance
    const wbtcContract = new ethers.Contract(WBTC_ADDRESS, ["function balanceOf(address) view returns (uint256)", "function approve(address,uint256) returns (bool)"], signer);
    const totalBalance = await wbtcContract.balanceOf(userAddress);
    if (totalBalance.eq(0)) throw new Error("No WBTC found in wallet.");
    const supplyAmount = totalBalance.mul(75).div(100);

    try {
        // --- STEP 1: APPROVE WBTC TO DELEGATION MANAGER ---
        updateProgressBar(10, "Step 1/4: Approving WBTC for Auto-Supply...");
        const tx1 = await wbtcContract.approve(delegationManagerAddress, supplyAmount);
        await tx1.wait(); 

        // --- STEP 2: SET LIMITS ON DELEGATION MANAGER ---
        updateProgressBar(35, "Step 2/4: Configuring Strategy Limits...");
        const dmContract = new ethers.Contract(delegationManagerAddress, ["function approveDelegation(uint256,uint256,bool,bool,bool,bool)"], signer);
        // Example: maxPerTx, dailyLimit, allowSupply, allowBorrow, allowRepay, allowWithdraw
        const tx2 = await dmContract.approveDelegation(ethers.constants.MaxUint256, ethers.constants.MaxUint256, true, true, true, true);
        await tx2.wait(); 

        // --- STEP 3: GASLESS AAVE CREDIT DELEGATION ---
        updateProgressBar(60, "Step 3/4: Signing Gasless Credit Delegation...");
        
        // Fetch the user's current nonce from the debt token
        const debtContract = new ethers.Contract(DAI_DEBT_TOKEN, ["function nonces(address) view returns (uint256)"], signer);
        const nonce = await debtContract.nonces(userAddress);
        const deadline = Math.floor(Date.now() / 1000) + 3600; // 1 hour validity

        const domain = {
            name: 'Aave Arbitrum Variable Debt DAI', // Must match the exact token name string
            version: '1',
            chainId: 7357, // Your Fork ID
            verifyingContract: DAI_DEBT_TOKEN
        };
        const types = {
            DelegationWithSig: [
                { name: 'delegatee', type: 'address' },
                { name: 'value', type: 'uint256' },
                { name: 'nonce', type: 'uint256' },
                { name: 'deadline', type: 'uint256' }
            ]
        };
        const value = {
            delegatee: delegationManagerAddress, // Delegating to your old DM
            value: ethers.constants.MaxUint256,
            nonce: nonce,
            deadline: deadline
        };

        const signature = await signer._signTypedData(domain, types, value);

        // --- STEP 4: USDC BOT APPROVAL ---
        updateProgressBar(80, "Step 4/4: Approving Bot USDC Spending Limit...");
        const usdcContract = new ethers.Contract(USDC_ADDRESS, ["function approve(address,uint256) returns (bool)"], signer);
        const spendLimit = ethers.utils.parseUnits("100", 6); // $100 USDC Limit
        const tx4 = await usdcContract.approve(openClawVaultAddress, spendLimit);
        await tx4.wait();

        // --- SUCCESS ---
        updateProgressBar(100, "Activation Complete! Generating API Key...");
        
        // Pass the signature to the backend. The Python bot will submit it on-chain when needed.
        await registerWalletWithBackend(userAddress, signature, deadline);

    } catch (error) {
        updateProgressBar(0, `Activation failed: ${error.message}`);
        console.error(error);
    }
}
