OBJECTIVE
We need to run BOTH the web_dashboard.py (Flask Server) and run_autonomous_mainnet.py (Trading Bot) simultaneously on the single Reserved VM.
Currently, the "Run" command only starts the dashboard, leaving the bot inactive.

TASK 1: Create the Execution Script
Action: Create a new file named run_both.py.

Content: Paste the following code exactly. It uses threading to run the agent in the background while the dashboard runs in the main thread (required for port binding).

Python
import threading
import subprocess
import time
import sys
import os

def run_agent():
    """Run the autonomous agent in a loop to handle crashes/restarts"""
    while True:
        try:
            print("🚀 Starting Autonomous Agent...", flush=True)
            # Use sys.executable to ensure we use the same python interpreter
            subprocess.run([sys.executable, "run_autonomous_mainnet.py"])
            print("⚠️ Agent stopped. Restarting in 5 seconds...", flush=True)
            time.sleep(5)
        except Exception as e:
            print(f"❌ Agent execution error: {e}. Restarting...", flush=True)
            time.sleep(5)

def run_dashboard():
    """Run the web dashboard in a loop"""
    while True:
        try:
            print("📊 Starting Web Dashboard...", flush=True)
            subprocess.run([sys.executable, "web_dashboard.py"])
            print("⚠️ Dashboard stopped. Restarting in 5 seconds...", flush=True)
            time.sleep(5)
        except Exception as e:
            print(f"❌ Dashboard execution error: {e}. Restarting...", flush=True)
            time.sleep(5)

if __name__ == "__main__":
    print("🤖 SYSTEM LAUNCH: Dual-Process Mode", flush=True)
    
    # 1. Start the Autonomous Agent in a background thread
    # daemon=True ensures it dies if the main program (dashboard) dies
    agent_thread = threading.Thread(target=run_agent, daemon=True)
    agent_thread.start()
    
    # 2. Run the Dashboard in the main thread (Blocking)
    # This keeps the script alive and satisfies the web server requirement
    run_dashboard()
