5. Collateral Velocity Monitor (Panic Detection)
Implement a Collateral Velocity Monitor as a separate module or class. It must:

Track a rolling 40‑minute history of total collateral value.

Every N seconds (e.g. 60s) compute:

drop_20 = value_20_minutes_ago - current

drop_30 = value_30_minutes_ago - current

Trigger:

Micro short when drop_20 >= 30.0

Macro short when drop_30 >= 50.0

Enforce cooldowns:

Micro: 4 hours between firings.

Macro: 12 hours between firings.

Call the existing trigger_micro_short() / trigger_macro_short() functions without changing their internal distribution logic.

Example structure:

python
# monitor.py
import time
from collections import deque
from .aave import get_total_collateral_usd
from .shorts import trigger_micro_short, trigger_macro_short

class CollateralVelocityMonitor:
    def __init__(self):
        self.history = deque()  # list of (timestamp, value_usd)
        self.micro_cooldown_seconds = 4 * 3600
        self.macro_cooldown_seconds = 12 * 3600
        self.last_micro_ts = 0
        self.last_macro_ts = 0

    def _record(self, now, value):
        self.history.append((now, value))
        cutoff = now - 40 * 60
        while self.history and self.history[0][0] < cutoff:
            self.history.popleft()

    def _value_at_or_before(self, target_ts):
        candidate = None
        for ts, val in self.history:
            if ts <= target_ts:
                candidate = (ts, val)
        return candidate[1] if candidate else None

    def step(self):
        now = int(time.time())
        current = get_total_collateral_usd()
        self._record(now, current)

        v20 = self._value_at_or_before(now - 20 * 60)
        v30 = self._value_at_or_before(now - 30 * 60)

        drop_20 = (v20 - current) if v20 is not None else 0
        drop_30 = (v30 - current) if v30 is not None else 0

        if drop_20 >= 30.0 and (now - self.last_micro_ts) >= self.micro_cooldown_seconds:
            trigger_micro_short()
            self.last_micro_ts = now

        if drop_30 >= 50.0 and (now - self.last_macro_ts) >= self.macro_cooldown_seconds:
            trigger_macro_short()
            self.last_macro_ts = now

import os
import httpx

PERPLEXITY_API_KEY = os.getenv("PERPLEXITY_API_KEY")

HEADERS = {
    "Authorization": f"Bearer {PERPLEXITY_API_KEY}",
    "Content-Type": "application/json",
}

async def perplexity_chat(system_prompt: str, user_prompt: str, model: str = "sonar-pro"):
    payload = {
        "model": model,
        "messages": [
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_prompt},
        ],
        "stream": False,
    }
    async with httpx.AsyncClient(timeout=60) as client:
        r = await client.post(PERPLEXITY_API_URL, headers=HEADERS, json=payload)
        r.raise_for_status()
        data = r.json()
        return data["choices"][0]["message"]["content"]
