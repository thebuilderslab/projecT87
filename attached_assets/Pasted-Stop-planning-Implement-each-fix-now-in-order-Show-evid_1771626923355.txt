def delegated_withdraw(user_address: str, asset: str, 
                       amount: float, asset_symbol: str) -> dict:
    """
    Withdraws asset from Aave on behalf of user.
    Aave sends withdrawn tokens to DelegationManager (msg.sender).
    We then forward them to the user via emergencyWithdrawToken.
    """
    try:
        # Step 1: Execute withdraw via DelegationManager
        withdraw_result = _execute_withdraw_on_chain(
            user_address, asset, amount
        )
        if not withdraw_result.get("success"):
            return {"success": False, 
                    "error": f"Withdraw failed: {withdraw_result}"}

        log.info(
            f"[DelegatedWithdraw] Step 1/3 OK: "
            f"withdrew {amount} {asset_symbol} for {user_address}"
        )

        # Step 2: Forward from DelegationManager to bot wallet
        # using emergencyWithdrawToken (owner-only)
        forward_to_bot = _emergency_withdraw_token(
            asset, amount
        )
        if not forward_to_bot.get("success"):
            log.error(
                f"[DelegatedWithdraw] Step 2/3 FAILED: "
                f"could not forward from DM to bot: {forward_to_bot}"
            )
            return {"success": False, 
                    "error": "Token forward DM→bot failed",
                    "step": 2}

        log.info(
            f"[DelegatedWithdraw] Step 2/3 OK: "
            f"forwarded {amount} {asset_symbol} DM→bot"
        )

        # Step 3: Transfer from bot wallet to user wallet
        transfer_result = _transfer_token_to_user(
            asset, user_address, amount
        )
        if not transfer_result.get("success"):
            log.error(
                f"[DelegatedWithdraw] Step 3/3 FAILED: "
                f"could not transfer bot→user: {transfer_result}"
            )
            return {"success": False, 
                    "error": "Token transfer bot→user failed",
                    "step": 3}

        log.info(
            f"[DelegatedWithdraw] Step 3/3 OK: "
            f"transferred {amount} {asset_symbol} bot→{user_address}"
        )

        return {"success": True, 
                "amount": amount, 
                "asset": asset_symbol,
                "steps_completed": 3}

    except Exception as e:
        log.error(f"[DelegatedWithdraw] Exception: {e}")
        return {"success": False, "error": str(e)}

Task 4: Fix nonce management in distribution sequence
The logs showed a "nonce too low" error during STEP4 of the distribution, caused by rapid sequential transactions using stale nonces.

Fix _send_bot_tx (or the nonce management layer) to fetch a fresh nonce before every transaction in the distribution sequence:

python
def _send_bot_tx(func_call, gas_limit=300000):
    """
    Builds, signs, and sends a bot transaction.
    Always fetches a fresh nonce to prevent 
    'nonce too low' errors in rapid sequences.
    """
    # Always fetch fresh nonce — never use cached
    nonce = w3.eth.get_transaction_count(
        BOT_WALLET, 
        "pending"  # use "pending" to account for 
                   # in-flight transactions
    )
    
    tx = func_call.build_transaction({
        "from": BOT_WALLET,
        "nonce": nonce,
        "gas": gas_limit,
        "gasPrice": w3.eth.gas_price,
        "chainId": 42161
    })
    
    signed = w3.eth.account.sign_transaction(
        tx, BOT_PRIVATE_KEY
    )
    tx_hash = w3.eth.send_raw_transaction(
        signed.rawTransaction
    )
    
    # Wait for confirmation before returning
    # This ensures nonce is consumed before 
    # next tx is built
    receipt = w3.eth.wait_for_transaction_receipt(
        tx_hash, timeout=60
    )
    
    return {
        "success": receipt.status == 1,
        "tx_hash": tx_hash.hex(),
        "gas_used": receipt.gasUsed
    }
Key changes:

Use "pending" block for nonce to account for in-flight transactions.

Wait for receipt before returning, so the caller cannot build the next transaction until the current one is confirmed.

This makes the distribution sequence fully sequential and prevents nonce collisions.
