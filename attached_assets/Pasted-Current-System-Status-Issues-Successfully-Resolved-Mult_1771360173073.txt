def create_document_in_users_drive(self, title, content, folder_id):
    """
    Create document via user's Drive storage, not service account's.
    This bypasses the service account storage quota issue.
    """
    try:
        # Create file metadata - file will use folder owner's quota
        file_metadata = {
            'name': title,
            'mimeType': 'application/vnd.google-apps.document',
            'parents': [folder_id]  # baileygroupservices@gmail.com owns this folder
        }
        
        # Create empty doc via Drive API
        file = self.drive_service.files().create(
            body=file_metadata,
            fields='id,webViewLink,name'
        ).execute()
        
        doc_id = file['id']
        
        # Add content via Docs API (now that file exists in user's storage)
        requests_body = {
            'requests': [{
                'insertText': {
                    'location': {'index': 1},
                    'text': content
                }
            }]
        }
        
        self.docs_service.documents().batchUpdate(
            documentId=doc_id,
            body=requests_body
        ).execute()
        
        logging.info(f"✅ Document created in user's Drive: {file['webViewLink']}")
        return {
            'id': doc_id,
            'url': file['webViewLink'],
            'name': file['name']
        }
        
    except Exception as e:
        if 'storageQuotaExceeded' in str(e):
            logging.error("❌ Folder owner's storage is full")

import asyncio
from concurrent.futures import ThreadPoolExecutor

async def run_multi_town_scrape_optimized():
    """Parallel scraping with intelligent error handling"""
    
    towns = [
        {"name": "Hartford", "url": "CTHAR", "priority": "high"},
        {"name": "East Hartford", "url": "CTEHART", "priority": "high"},
        {"name": "Windsor", "url": "CTWSR", "priority": "medium"},
        {"name": "Berlin", "url": "CTBER", "priority": "low"},
        {"name": "Rocky Hill", "url": "CTROCK", "priority": "low"}
    ]
    
    results = {}
    
    # Parallel execution with rate limiting
    with ThreadPoolExecutor(max_workers=3) as executor:
        futures = {}
        
        for town in towns:
            # Stagger requests by 2 seconds
            time.sleep(2)
            
            future = executor.submit(
                scrape_town_with_retry,
                town['name'],
                town['url'],
                max_retries=3
            )
            futures[future] = town['name']
        
        # Collect results as they complete
        for future in as_completed(futures):
            town_name = futures[future]
            try:
                town_results = future.result(timeout=60)
                results[town_name] = town_results
                logging.info(f"✅ {town_name}: {len(town_results)} filings")
            except Exception as e:
                logging.error(f"❌ {town_name} failed: {e}")
                results[town_name] = []
    
    return results

def scrape_town_with_retry(town_name, base_code, max_retries=3):
    """Scrape with exponential backoff retry"""
    base_url = f"https://www.searchiqs.com/{base_code}/"
    
    for attempt in range(max_retries):
        try:
            scraper = SearchIQSScraper(base_url, town_name)
            filings = scraper.search_lis_pendens(
                days_back=30,
                fetch_details=True  # Get full details in one pass
            )
            
            # Validate data quality
            valid_filings = [
                f for f in filings 
                if f.get('party1_seller') and f.get('recording_date')
            ]
            
            if len(valid_filings) < len(filings):
                logging.warning(
                    f"{town_name}: {len(filings) - len(valid_filings)} "
                    f"filings missing required fields"
                )
            
            return valid_filings
            
        except Exception as e:
            wait_time = 2 ** attempt  # Exponential backoff: 1s, 2s, 4s
            logging.warning(
                f"{town_name} attempt {attempt+1} failed: {e}. "
                f"Retrying in {wait_time}s..."
            )
            time.sleep(wait_time)
    
    # All retries exhausted
    logging.error(f"{town_name}: All {max_retries} attempts failed")
    return []

def process_and_enrich_filings(all_filings):
    """Clean, deduplicate, and enrich filing data"""
    
    # 1. Deduplicate across towns
    unique_filings = {}
    for town, filings in all_filings.items():
        for filing in filings:
            # Create unique key
            key = f"{filing.get('book_page')}_{filing.get('recording_date')}"
            
            if key not in unique_filings:
                filing['towns'] = [town]
                unique_filings[key] = filing
            else:
                # Same filing in multiple towns
                unique_filings[key]['towns'].append(town)
    
    # 2. Data quality scoring
    for filing in unique_filings.values():
        score = 0
        score += 30 if filing.get('property_address') else 0
        score += 20 if filing.get('party1_seller') else 0
        score += 20 if filing.get('party2_plaintiff') else 0
        score += 15 if filing.get('recording_date') else 0
        score += 15 if filing.get('book_page') else 0
        
        filing['data_quality_score'] = score
    
    # 3. Priority classification
    for filing in unique_filings.values():
        filing['priority'] = classify_priority(filing)
    
    return list(unique_filings.values())

def classify_priority(filing):
    """Intelligent priority classification"""
    priority_score = 0
    
    # Recent filings = higher priority
    try:
        filing_date = datetime.strptime(
            filing.get('recording_date', ''),
            '%m/%d/%Y'
        )
        days_old = (datetime.now() - filing_date).days
        
        if days_old <= 7:
            priority_score += 40
        elif days_old <= 14:
            priority_score += 25
        elif days_old <= 30:
            priority_score += 10
    except:
        pass
    
    # Complete property info = higher priority
    if filing.get('property_address'):
        priority_score += 30
    
    # Multiple party names = more complex case = higher priority
    if filing.get('party1_seller') and filing.get('party2_plaintiff'):
        priority_score += 20
    
    # Data quality bonus
    priority_score += filing.get('data_quality_score', 0) * 0.1
    
    # Classify
    if priority_score >= 70:
        return 'HIGH'
    elif priority_score >= 40:
        return 'MEDIUM'
    else:
        return 'LOW'
