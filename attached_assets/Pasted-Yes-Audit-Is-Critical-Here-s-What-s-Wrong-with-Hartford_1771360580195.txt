import requests
from bs4 import BeautifulSoup
import json

def audit_search_form(base_url, town_name):
    """Extract and compare search form structures"""
    
    session = requests.Session()
    
    # Step 1: Login as guest
    login_url = f"{base_url}/Default.aspx"
    response = session.get(login_url)
    soup = BeautifulSoup(response.text, 'html.parser')
    
    viewstate = soup.find('input', {'name': '__VIEWSTATE'})['value']
    
    # Guest login
    login_data = {
        '__EVENTTARGET': 'btnGuestLogin',
        '__VIEWSTATE': viewstate,
    }
    session.post(login_url, data=login_data)
    
    # Step 2: Get search page
    search_url = f"{base_url}/SearchAdvancedMP.aspx"
    response = session.get(search_url)
    soup = BeautifulSoup(response.text, 'html.parser')
    
    # Extract form structure
    doc_group = soup.find('select', {'id': 'ContentPlaceHolder1_cboDocGroup'})
    doc_type = soup.find('select', {'id': 'ContentPlaceHolder1_cboDocType'})
    
    audit_data = {
        'town': town_name,
        'url': base_url,
        'doc_groups': [],
        'doc_types': [],
        'has_town_dropdown': bool(soup.find('select', {'id': 'ContentPlaceHolder1_cboTown'})),
        'has_ignore_party_checkbox': bool(soup.find('input', {'id': 'ContentPlaceHolder1_chkIgnorePartyType'}))
    }
    
    # Get all doc group options
    if doc_group:
        for option in doc_group.find_all('option'):
            audit_data['doc_groups'].append({
                'value': option.get('value', ''),
                'text': option.text.strip()
            })
    
    # Get all doc type options
    if doc_type:
        for option in doc_type.find_all('option'):
            if 'LIS PENDENS' in option.text.upper() or 'FORECLOSURE' in option.text.upper():
                audit_data['doc_types'].append({
                    'value': option.get('value', ''),
                    'text': option.text.strip()
                })
    
    return audit_data

def _get_town_specific_params(self, base_params):
    """
    Apply town-specific search parameter overrides
    Different towns require different cboDocGroup settings
    """
    
    town_overrides = {
        'CTHAR': {  # Hartford - REQUIRES explicit LAND RECORDS group
            'ctl00$ContentPlaceHolder1$cboDocGroup': 'LAND RECORDS',
            'reason': 'Hartford requires explicit LAND RECORDS group, not (ALL)'
        },
        'CTEHART': {  # East Hartford - works with (ALL)
            'ctl00$ContentPlaceHolder1$cboDocGroup': '(ALL)',
        },
        'CTWSR': {  # Windsor - works with (ALL)
            'ctl00$ContentPlaceHolder1$cboDocGroup': '(ALL)',
        },
        'CTBER': {  # Berlin - works with (ALL)
            'ctl00$ContentPlaceHolder1$cboDocGroup': '(ALL)',
        },
        'CTROCK': {  # Rocky Hill - may need explicit group
            'ctl00$ContentPlaceHolder1$cboDocGroup': '(ALL)',
        }
    }
    
    # Detect town code from base_url
    town_code = None
    for code in town_overrides.keys():
        if code in self.base_url:
            town_code = code
            break
    
    if town_code and town_code in town_overrides:
        override = town_overrides[town_code]
        logging.info(f"🔧 Applying {town_code}-specific parameters")
        
        if 'reason' in override:
            logging.info(f"   Reason: {override['reason']}")
        
        # Apply overrides
        for key, value in override.items():
            if key != 'reason':
                base_params[key] = value
                logging.debug(f"   Override: {key} = {value}")
    
    return base_params
