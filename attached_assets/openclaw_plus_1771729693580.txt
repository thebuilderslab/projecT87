pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/IERC20.sol";

contract OpenClawVault {
    IERC20 public usdc;
    address public owner; // The Python API's wallet address

    constructor(address _usdcAddress) {
        usdc = IERC20(_usdcAddress); 
        owner = msg.sender;
    }

    modifier onlyOwner() {
        require(msg.sender == owner, "Not authorized by OpenClaw API");
        _;
    }

    /**
     * @dev Called by the Python backend when a gift card/NFT is purchased.
     * Pulls the exact USDC amount from the user and routes it to the destination.
     */
    function executeTransfer(address userWallet, address destinationWallet, uint256 amount) external onlyOwner {
        require(usdc.transferFrom(userWallet, destinationWallet, amount), "USDC transfer failed. Check user allowance.");
    }
    
    /**
     * @dev Allows the API to change the authorized operational wallet if needed.
     */
    function transferOwnership(address newOwner) external onlyOwner {
        require(newOwner != address(0), "Invalid address");
        owner = newOwner;
    }
}
