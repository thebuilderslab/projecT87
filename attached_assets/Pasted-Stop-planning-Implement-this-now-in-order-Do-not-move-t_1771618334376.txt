Step 3: Build the authorization script
Based on the contract source findings from Step 1, write a script that the deployer wallet (0xbbd55BB1...7e27C9c48) can run to authorize the bot wallet as botOperator:

python
# scripts/set_bot_operator.py
# Run ONCE from the deployer wallet to authorize the bot wallet
# as botOperator on REAADelegationManager

import os
from web3 import Web3
from eth_account import Account

w3 = Web3(Web3.HTTPProvider(os.environ["RPC_URL"]))

CONTRACT_ADDRESS = "0x7427370Ab4C311B090446544078c819b3946E59d"
DEPLOYER_PRIVATE_KEY = os.environ["DEPLOYER_PRIVATE_KEY"]
BOT_WALLET = Account.from_key(
    os.environ["BOT_PRIVATE_KEY"]
).address

deployer = Account.from_key(DEPLOYER_PRIVATE_KEY)

contract = w3.eth.contract(
    address=CONTRACT_ADDRESS,
    abi=DELEGATION_MANAGER_ABI
)

# Confirm current state before making changes
current_operator = contract.functions.botOperator().call()
owner = contract.functions.owner().call()

print(f"Current botOperator : {current_operator}")
print(f"Contract owner      : {owner}")
print(f"Deployer wallet     : {deployer.address}")
print(f"Bot wallet to set   : {BOT_WALLET}")
print()

# Safety check — confirm deployer is owner
assert owner.lower() == deployer.address.lower(), \
    f"ERROR: Deployer {deployer.address} is not the contract owner {owner}"

# Build the transaction
# Replace setBotOperator with the actual function name from Step 1
tx = contract.functions.setBotOperator(BOT_WALLET).build_transaction({
    "from": deployer.address,
    "nonce": w3.eth.get_transaction_count(deployer.address),
    "gas": 100000,
    "gasPrice": w3.eth.gas_price,
    "chainId": 42161  # Arbitrum mainnet
})

print(f"Transaction built. Gas estimate: {tx['gas']}")
print(f"Setting botOperator to: {BOT_WALLET}")
print()

# Sign and send
signed = w3.eth.account.sign_transaction(tx, DEPLOYER_PRIVATE_KEY)
tx_hash = w3.eth.send_raw_transaction(signed.rawTransaction)
print(f"Transaction sent: {tx_hash.hex()}")

# Wait for confirmation
receipt = w3.eth.wait_for_transaction_receipt(tx_hash, timeout=120)
print(f"Status: {'SUCCESS' if receipt.status == 1 else 'FAILED'}")
print(f"Block: {receipt.blockNumber}")
print(f"Gas used: {receipt.gasUsed}")

# Verify
new_operator = contract.functions.botOperator().call()
print(f"New botOperator: {new_operator}")
print(f"Correctly set  : "
      f"{new_operator.lower() == BOT_WALLET.lower()}")
