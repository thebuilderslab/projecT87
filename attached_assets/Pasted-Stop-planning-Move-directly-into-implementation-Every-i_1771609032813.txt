python
def validate_full_automation_ready(user_id: int, wallet_address: str) -> dict:
    blockers = []

    # Check delegation flags
    delegation = contract.functions.getDelegation(wallet_address).call()
    flags = {
        "allowSupply": delegation[2],
        "allowBorrow": delegation[3],
        "allowRepay": delegation[4],
        "allowWithdraw": delegation[5],
    }
    missing_flags = [k for k, v in flags.items() if not v]
    if missing_flags:
        blockers.append({
            "type": "delegation_flags",
            "missing": missing_flags
        })

    # Check ERC20 approvals (5 tokens x 3 contracts)
    tokens = {
        "DAI": DAI_ADDRESS, "WETH": WETH_ADDRESS, "WBTC": WBTC_ADDRESS,
        "USDC": USDC_ADDRESS, "USDT": USDT_ADDRESS
    }
    spenders = {
        "DelegationManager": DELEGATION_MANAGER,
        "AavePool": AAVE_POOL,
        "UniswapRouter": UNISWAP_ROUTER
    }
    for token_name, token_addr in tokens.items():
        for spender_name, spender_addr in spenders.items():
            allowance = get_erc20_allowance(token_addr, wallet_address,
                                            spender_addr)
            if allowance < MIN_REQUIRED_ALLOWANCE:
                blockers.append({
                    "type": "erc20_approval",
                    "token": token_name,
                    "spender": spender_name,
                    "current_allowance": str(allowance)
                })

    ready = len(blockers) == 0
    if ready:
        database.update_strategy_status_field(user_id, wallet_address,
                                              'active')
        database.set_bot_enabled(user_id, True)
        log.info(f"[Activation] Wallet {wallet_address} fully validated "
                 f"— strategy_status=active")
    else:
        database.update_strategy_status_field(user_id, wallet_address,
                                              'error_permissions')
        log.warning(f"[Activation] Wallet {wallet_address} NOT ready — "
                    f"blockers: {blockers}")

    return {"ready": ready, "blockers": blockers}
