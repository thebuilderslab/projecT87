import os
import difflib
import shutil
import subprocess
from collections import defaultdict
import re

# Keywords to group files by functionality
KEYWORDS = [
    "diagnostic", "compliance", "launcher", "dashboard",
    "sync", "test", "emergency", "health_monitor", "fetcher"
]

BASE_DIR = "."
REPORT_FILE = "keep_remove_report.txt"
ARCHIVE_DIR = "archive_duplicates"

def scan_files():
    py_files = []
    for root, _, files in os.walk(BASE_DIR):
        for f in files:
            if f.endswith(".py"):
                py_files.append(os.path.join(root, f))
    return py_files

def group_by_keyword(files):
    groups = defaultdict(list)
    for f in files:
        name_lower = os.path.basename(f).lower()
        matched = False
        for kw in KEYWORDS:
            if kw in name_lower:
                groups[kw].append(f)
                matched = True
                break
        if not matched:
            groups["misc"].append(f)
    return groups

def git_latest_commit_date(file_path):
    try:
        result = subprocess.run(
            ["git", "log", "-1", "--format=%ct", "--", file_path],
            capture_output=True, text=True, check=True)
        timestamp = int(result.stdout.strip())
        return timestamp
    except Exception:
        # If Git info unavailable, fallback to file mod time
        return int(os.path.getmtime(file_path))

def choose_canonical(files):
    # Choose file by latest git commit timestamp (or fallback mod time)
    return sorted(files, key=git_latest_commit_date, reverse=True)[0]

def merge_missing_code(canonical, old_file):
    with open(canonical, "r", encoding="utf-8") as f1, open(old_file, "r", encoding="utf-8") as f2:
        canonical_lines = f1.readlines()
        old_lines = f2.readlines()

    # Simple merge approach: add lines from old file that are not in canonical,
    # ignoring empty lines and duplicates.
    additions = [line for line in old_lines if line.strip() and line not in canonical_lines]

    if additions:
        with open(canonical, "a", encoding="utf-8") as f:
            f.write("\n# --- Merged from older version: {} ---\n".format(os.path.basename(old_file)))
            f.writelines(additions)

def archive_file(file_path):
    os.makedirs(ARCHIVE_DIR, exist_ok=True)
    dst = os.path.join(ARCHIVE_DIR, os.path.basename(file_path))
    counter = 1
    base_dst = dst
    # Avoid overwriting archive files
    while os.path.exists(dst):
        dst = f"{base_dst}_{counter}"
        counter += 1
    shutil.move(file_path, dst)

def refactor_imports(canonical_map):
    # For each archived file and its canonical replacement,
    # search project files and replace imports referencing old file with canonical file.
    pattern_template = r"(\bfrom\s+)([\w\.]+)(\s+import\s+.+)"  # from module import something
    import_template = r"(\bimport\s+)([\w\.]+)"                # import module

    for root, _, files in os.walk(BASE_DIR):
        for f in files:
            if f.endswith(".py"):
                path = os.path.join(root, f)
                with open(path, 'r', encoding='utf-8') as file:
                    content = file.read()

                original_content = content

                for old_file, canonical_file in canonical_map.items():
                    old_mod = os.path.splitext(os.path.basename(old_file))[0]
                    canonical_mod = os.path.splitext(os.path.basename(canonical_file))[0]

                    # Replace 'from old_module import' and 'import old_module'
                    # Use word boundaries to avoid partial matches
                    content = re.sub(rf'\bfrom\s+{re.escape(old_mod)}\b', f'from {canonical_mod}', content)
                    content = re.sub(rf'\bimport\s+{re.escape(old_mod)}\b', f'import {canonical_mod}', content)

                if content != original_content:
                    with open(path, 'w', encoding='utf-8') as file:
                        file.write(content)

def main():
    files = scan_files()
    groups = group_by_keyword(files)
    report_lines = []
    canonical_map = {}  # Map old_file -> canonical_file for import refactoring

    for kw, group_files in groups.items():
        if len(group_files) < 2:
            continue  # No duplicates to process

        canonical = choose_canonical(group_files)
        report_lines.append(f"\n[{kw.upper()}] Canonical: {canonical}")

        for f in group_files:
            if f != canonical:
                merge_missing_code(canonical, f)
                archive_file(f)
                canonical_map[f] = canonical
                report_lines.append(f"  Archive: {f} (merged into canonical)")

    refactor_imports(canonical_map)

    with open(REPORT_FILE, "w", encoding="utf-8") as rf:
        rf.write("\n".join(report_lines))

    print(f"✅ Deduplication complete. See {REPORT_FILE} and {ARCHIVE_DIR}/ for results.")

if __name__ == "__main__":
    main()
